    // find the user

Easy Authentication

	
								
Route::post('login', function()
{
    $credentials = array(
        'username' => Input::get('username'),
        'password' => Input::get('password')
    );
 
    // perform validation
 
    if ( Auth::attempt($credentials) )
    {
        // Credentials match. Logged in!
        return Redirect::to('admin/profile');
    }
});
